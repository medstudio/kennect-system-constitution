# Storage in 3rd party Storage array will contain objects as specified below

MONGO STORAGE CONFIG

{
    "type": "mongo",
    "uri": {
        "primary": true,
        "dbms": "mongodb",
        "host": "localhost:27017",
        "dbName": "warehouse_data",
        "pw": false
    },
    "collection_names": [
        "commission"
    ]
}
